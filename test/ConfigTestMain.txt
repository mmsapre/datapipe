package com.mas.sap.prak.client360;

import com.fasterxml.jackson.databind.ObjectMapper;
import com.fasterxml.jackson.dataformat.yaml.YAMLFactory;
import com.mas.sap.prak.client360.config.PipelineConfig;

import java.io.File;

public class ConfigTestMain {

    public static void main(String[] args)
            throws Exception {

        ObjectMapper mapper =
                new ObjectMapper(
                        new YAMLFactory());

        PipelineConfig config =
                mapper.readValue(
                        new File(
                                "src/main/resources/client360.yml"),
                        PipelineConfig.class);

        System.out.println(
                "Sources="
                + config.getSources().size());

        config.getSources()
                .forEach(
                        s -> System.out.println(
                                s.getEntityType()
                                + " -> "
                                + s.getFileName()));
    }
}
