

import java.util.ArrayList;
import java.util.Collections;
import java.util.List;
import java.util.Map;

public final class SourceConfigMapper {

    private SourceConfigMapper() {
    }

    public static List<MrdSourceConfig> map(
            List<Map<String,Object>> sourceMaps) {

        if(sourceMaps == null) {
            return Collections.emptyList();
        }

        List<MrdSourceConfig> sources =
                new ArrayList<>();

        for(Map<String,Object> map : sourceMaps) {

            MrdSourceConfig cfg =
                    new MrdSourceConfig();

            cfg.setSourceName(
                    stringValue(
                            map,
                            "source-name"));

            cfg.setFileName(
                    stringValue(
                            map,
                            "file-name"));

            cfg.setEntityType(
                    stringValue(
                            map,
                            "entity-type"));

            cfg.setEntityIdField(
                    stringValue(
                            map,
                            "entity-id-field"));

            cfg.setCanonicalNameField(
                    stringValue(
                            map,
                            "canonical-name-field"));

            cfg.setBlockingFields(
                    stringList(
                            map,
                            "blocking-fields"));

            cfg.setSearchFields(
                    stringList(
                            map,
                            "search-fields"));

            cfg.setDisplayFields(
                    stringList(
                            map,
                            "display-fields"));

            cfg.setMatchingFields(
                    stringList(
                            map,
                            "matching-fields"));

            cfg.setRelationFields(
                    stringList(
                            map,
                            "relation-fields"));

            cfg.setRules(
                    mapRules(map));

            sources.add(cfg);
        }

        return sources;
    }

    @SuppressWarnings("unchecked")
    private static List<MrdMatchRule> mapRules(
            Map<String,Object> sourceMap) {

        Object obj =
                sourceMap.get("rules");

        if(obj == null) {
            return Collections.emptyList();
        }

        List<Map<String,Object>> maps =
                (List<Map<String,Object>>) obj;

        List<MrdMatchRule> rules =
                new ArrayList<>();

        for(Map<String,Object> ruleMap : maps) {

            MrdMatchRule rule =
                    new MrdMatchRule();

            rule.setField(
                    stringValue(
                            ruleMap,
                            "field"));

            rule.setAlgorithm(
                    stringValue(
                            ruleMap,
                            "algorithm"));

            rule.setWeight(
                    doubleValue(
                            ruleMap,
                            "weight"));

            rules.add(rule);
        }

        return rules;
    }

    @SuppressWarnings("unchecked")
    private static List<String> stringList(
            Map<String,Object> map,
            String key) {

        Object value =
                map.get(key);

        if(value == null) {
            return Collections.emptyList();
        }

        return (List<String>) value;
    }

    private static String stringValue(
            Map<String,Object> map,
            String key) {

        Object value =
                map.get(key);

        return value == null
                ? null
                : value.toString();
    }

    private static double doubleValue(
            Map<String,Object> map,
            String key) {

        Object value =
                map.get(key);

        if(value == null) {
            return 0;
        }

        return Double.parseDouble(
                value.toString());
    }
}
