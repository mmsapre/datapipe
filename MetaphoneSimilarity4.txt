package com.mas.sap.prak.client360.matching;

import org.apache.commons.codec.language.DoubleMetaphone;

public class MetaphoneSimilarity
        implements SimilarityAlgorithm {

    private final
    DoubleMetaphone metaphone =
            new DoubleMetaphone();

    @Override
    public ComparatorType getType() {

        return ComparatorType.METAPHONE;
    }

    @Override
    public double score(
            String left,
            String right) {

        if(left == null ||
           right == null) {

            return 0.0;
        }

        String l =
                metaphone.doubleMetaphone(
                        left);

        String r =
                metaphone.doubleMetaphone(
                        right);

        return l.equals(r)
                ? 1.0
                : 0.0;
    }
}
