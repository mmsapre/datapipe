package com.mas.sap.prak.client360.candidate;

import com.mas.sap.prak.client360.config.ExportConfig;
import org.apache.commons.csv.CSVFormat;
import org.apache.commons.csv.CSVPrinter;

import java.io.File;
import java.io.FileWriter;
import java.util.List;

public class CandidatePairExporter {

    private final ExportConfig config;

    public CandidatePairExporter(
            ExportConfig config) {

        this.config = config;
    }

    public void export(
            List<CandidatePair> pairs)
            throws Exception {

        File file =
                new File(
                        config.getDirectory(),
                        "candidate_pairs.csv");

        try(CSVPrinter printer =
                    new CSVPrinter(
                            new FileWriter(file),
                            CSVFormat.DEFAULT.withHeader(
                                    "left_entity_id",
                                    "right_entity_id",
                                    "left_entity_type",
                                    "right_entity_type",
                                    "block_key",
                                    "block_hint"))) {

            for(CandidatePair pair :
                    pairs) {

                printer.printRecord(
                        pair.getLeftEntityId(),
                        pair.getRightEntityId(),
                        pair.getLeftEntityType(),
                        pair.getRightEntityType(),
                        pair.getBlockKey(),
                        pair.getBlockHint());
            }
        }

        System.out.println(
                "Candidate pairs exported : "
                + file.getAbsolutePath());
    }
}
