package com.mas.sap.prak.client360.matching;

public class ExactSimilarity
        implements SimilarityAlgorithm {

    @Override
    public ComparatorType getType() {

        return ComparatorType.EXACT;
    }

    @Override
    public double score(
            String left,
            String right) {

        if(left == null ||
           right == null) {

            return 0.0;
        }

        return left.trim()
                   .equalsIgnoreCase(
                           right.trim())
                ? 1.0
                : 0.0;
    }
}
