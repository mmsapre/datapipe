package com.mas.sap.prak.client360.matching;

import com.mas.sap.prak.client360.config.SemanticConfig;

import java.net.URI;
import java.net.http.HttpClient;
import java.net.http.HttpRequest;
import java.net.http.HttpResponse;
import java.nio.charset.StandardCharsets;

public class SemanticSimilarity implements SimilarityAlgorithm {

    private final SemanticConfig config;
    private final HttpClient client = HttpClient.newHttpClient();
    private final JaroWinklerSimilarity fallback = new JaroWinklerSimilarity();

    public SemanticSimilarity(SemanticConfig config) {
        this.config = config;
    }

    @Override
    public double score(String left, String right) {

        if (left == null || right == null) {
            return 0.0;
        }

        if (config == null || !config.isEnabled()) {
            return fallback.score(left, right);
        }

        try {
            String body =
                    "{"
                            + "\"model\":\"" + config.getEmbeddingModel() + "\","
                            + "\"left\":\"" + escape(left) + "\","
                            + "\"right\":\"" + escape(right) + "\""
                            + "}";

            HttpRequest.Builder builder =
                    HttpRequest.newBuilder()
                            .uri(URI.create(config.getLlmUrl() + "/similarity"))
                            .header("Content-Type", "application/json")
                            .POST(HttpRequest.BodyPublishers.ofString(
                                    body,
                                    StandardCharsets.UTF_8));

            if (config.getLlmToken() != null
                    && !config.getLlmToken().isBlank()) {
                builder.header("Authorization", "Bearer " + config.getLlmToken());
            }

            HttpResponse<String> response =
                    client.send(
                            builder.build(),
                            HttpResponse.BodyHandlers.ofString());

            if (response.statusCode() >= 200
                    && response.statusCode() < 300) {

                return Double.parseDouble(
                        response.body().trim());
            }

        } catch (Exception ignored) {
        }

        return fallback.score(left, right);
    }

    private String escape(String value) {
        return value.replace("\"", "\\\"");
    }
}
