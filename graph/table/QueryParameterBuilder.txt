

import java.util.*;

public class QueryParameterBuilder {

    public List<Object> build(
            MrdConnectorConfig connector) {

        List<Object> params =
                new ArrayList<>();

        if(connector.getWhere() == null) {
            return params;
        }

        for(MrdWhereConfig where :
                connector.getWhere()) {

            if("IN".equals(
                    where.getOperator())) {

                params.addAll(
                        where.getValues());
            }
            else {

                params.add(
                        where.getValue());
            }
        }

        return params;
    }
}
