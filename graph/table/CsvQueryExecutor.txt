package com.mas.sap.prak.client360.executor;

import com.mas.sap.prak.client360.config.*;
import com.mas.sap.prak.client360.model.ImportedRecord;
import org.apache.commons.csv.*;
import org.springframework.stereotype.Component;

import java.io.*;
import java.nio.file.*;
import java.util.*;
import java.util.concurrent.*;

@Component
public class CsvQueryExecutor implements QueryExecutor {
    private final Client360Properties properties;
    private final Map<String, List<ImportedRecord>> cache = new ConcurrentHashMap<>();

    public CsvQueryExecutor(Client360Properties properties) {
        this.properties = properties;
    }

    @Override
    public List<ImportedRecord> fetch(NodeConfig node, String field, Object value) throws Exception {
        List<ImportedRecord> rows = load(node);
        List<ImportedRecord> result = new ArrayList<>();

        for (ImportedRecord r : rows) {
            Object actual = r.get(field);
            if (actual != null && String.valueOf(actual).equals(String.valueOf(value))) {
                result.add(r);
            }
        }

        return result;
    }

    private List<ImportedRecord> load(NodeConfig node) throws Exception {
        return cache.computeIfAbsent(node.getEntityType(), k -> {
            try {
                DataSourceConfig ds = properties.getDatasources().get(node.getConnector().getDatasource());
                Path path = Paths.get(ds.getLocation(), node.getConnector().getSourceObject());

                try (Reader reader = Files.newBufferedReader(path);
                     CSVParser parser = CSVFormat.DEFAULT
                             .builder()
                             .setHeader()
                             .setSkipHeaderRecord(true)
                             .build()
                             .parse(reader)) {

                    List<ImportedRecord> out = new ArrayList<>();

                    for (CSVRecord row : parser) {
                        ImportedRecord r = new ImportedRecord();
                        r.setEntityType(node.getEntityType());
                        r.setLogicalEntityId(row.get(node.getConnector().getEntityIdField()));

                        for (String c : node.requiredColumns()) {
                            String v = row.isMapped(c) ? row.get(c) : null;
                            r.put(c, v);
                            if (node.getSearchFields().contains(c)) r.getSearchFields().put(c, v);
                            if (node.getDisplayFields().contains(c)) r.getDisplayFields().put(c, v);
                        }

                        out.add(r);
                    }

                    return out;
                }
            } catch (Exception e) {
                throw new RuntimeException(e);
            }
        });
    }
}
