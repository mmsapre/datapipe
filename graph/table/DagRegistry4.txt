package com.hsbc.gbm.hss.mrd.client360.registry;

import com.hsbc.gbm.hss.mrd.client360.config.MrdClient360Config;
import com.hsbc.gbm.hss.mrd.client360.config.MrdClient360NodeConfig;

import java.util.Collection;
import java.util.LinkedHashMap;
import java.util.Map;

public class DagRegistry {

    private final Map<String,RuntimeNode>
            nodes =
            new LinkedHashMap<>();

    public DagRegistry(
            MrdClient360Config config,
            DatasourceRegistry datasourceRegistry) {

        for(MrdClient360NodeConfig node :
                config.getDag().getNodes()) {

            String datasourceAlias =
                    node.getConnector()
                        .getDatasource();

            RuntimeNode runtimeNode =
                    new RuntimeNode(
                            node,
                            datasourceRegistry.get(
                                    datasourceAlias));

            nodes.put(
                    node.getEntityType(),
                    runtimeNode);
        }
    }

    public RuntimeNode get(
            String entityType) {

        return nodes.get(entityType);
    }

    public Collection<RuntimeNode> all() {

        return nodes.values();
    }
}
