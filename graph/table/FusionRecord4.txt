

import java.util.*;

public class FusionRecord {

    private String fusionId;

    private String rootEntity;

    private Map<String,Object>
            attributes =
            new LinkedHashMap<>();

    private List<String>
            participatingNodes =
            new ArrayList<>();

    public void addAttribute(
            String name,
            Object value) {

        attributes.put(
                name,
                value);
    }
}
