package com.mas.sap.prak.client360;

import com.mas.sap.prak.client360.config.Client360Properties;
import com.mas.sap.prak.client360.dag.DagValidator;
import org.springframework.boot.*;
import org.springframework.boot.autoconfigure.SpringBootApplication;
import org.springframework.boot.context.properties.EnableConfigurationProperties;

@SpringBootApplication
@EnableConfigurationProperties(Client360Properties.class)
public class Client360Application implements CommandLineRunner {
    private final DagValidator validator;

    public Client360Application(DagValidator validator) {
        this.validator = validator;
    }

    public static void main(String[] args) {
        SpringApplication.run(Client360Application.class, args);
    }

    @Override
    public void run(String... args) {
        validator.validate();
    }
}
