

import org.apache.ignite.IgniteCache;
import org.apache.ignite.cache.query.SqlFieldsQuery;

import java.util.ArrayList;
import java.util.List;

public class IgniteSearchService {

    private final IgniteCache<String, IgniteSearchDocument> cache;

    public IgniteSearchService(
            IgniteCache<String, IgniteSearchDocument> cache) {

        this.cache = cache;
    }

    public List<IgniteSearchDocument> searchLike(
            String text) {

        SqlFieldsQuery query =
                new SqlFieldsQuery(
                        """
                        select _key, searchText
                        from IgniteSearchDocument
                        where activeFlg = 'Y'
                        and searchText like ?
                        """);

        query.setArgs(
                "%"
                        + text.toLowerCase()
                        + "%");

        List<IgniteSearchDocument> result =
                new ArrayList<>();

        for (List<?> row :
                cache.query(query).getAll()) {

            String key =
                    String.valueOf(row.get(0));

            IgniteSearchDocument doc =
                    cache.get(key);

            if (doc != null) {
                result.add(doc);
            }
        }

        return result;
    }
}
