package com.hsbc.gbm.hss.mrd.client360.model;

import java.util.LinkedHashMap;
import java.util.Map;

public class ImportedRecord {

    private String entityType;

    private String logicalEntityId;

    private final Map<String,Object> values =
            new LinkedHashMap<>();

    public Object get(String field) {
        return values.get(field);
    }

    public void put(
            String field,
            Object value) {

        values.put(field, value);
    }

    public Map<String,Object> getValues() {
        return values;
    }

    public String getEntityType() {
        return entityType;
    }

    public void setEntityType(
            String entityType) {
        this.entityType = entityType;
    }

    public String getLogicalEntityId() {
        return logicalEntityId;
    }

    public void setLogicalEntityId(
            String logicalEntityId) {
        this.logicalEntityId = logicalEntityId;
    }
}
