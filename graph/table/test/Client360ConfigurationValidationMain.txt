package com.hsbc.gbm.hss.mrd.client360.test;

import com.hsbc.gbm.hss.mrd.client360.config.*;
import com.hsbc.gbm.hss.mrd.client360.registry.*;
import com.hsbc.gbm.hss.mrd.client360.validation.*;

public class Client360ConfigurationValidationMain {

    public static void main(String[] args)
            throws Exception {

        MrdClient360Config config =
                Client360YamlLoader.load(
                        "client360.yml");

        DatasourceRegistry dsRegistry =
                new DatasourceRegistry(
                        config);

        DagRegistry dagRegistry =
                new DagRegistry(
                        config,
                        dsRegistry);

        ConfigurationValidator validator =
                new ConfigurationValidator();

        validator.validate(
                config,
                dsRegistry,
                dagRegistry);

        System.out.println(
                "CONFIG VALIDATION SUCCESS");
    }
}
