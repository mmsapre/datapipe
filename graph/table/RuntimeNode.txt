package com.hsbc.gbm.hss.mrd.client360.dag;

public class RuntimeNode {

    private MrdClient360NodeConfig node;

    private MrdDatasourceConfig datasource;

    public RuntimeNode(
            MrdClient360NodeConfig node,
            MrdDatasourceConfig datasource) {

        this.node = node;
        this.datasource = datasource;
    }

    public MrdClient360NodeConfig getNode() {
        return node;
    }

    public MrdDatasourceConfig getDatasource() {
        return datasource;
    }
}
