package com.hsbc.gbm.hss.mrd.client360.graph;

import java.util.*;

public class GraphNode {

    private String id;

    private String label;

    private Map<String,Object> properties =
            new LinkedHashMap<>();

    public GraphNode() {
    }

    public GraphNode(
            String id,
            String label) {

        this.id = id;
        this.label = label;
    }

    public String getId() {
        return id;
    }

    public String getLabel() {
        return label;
    }

    public Map<String,Object>
    getProperties() {
        return properties;
    }
}
