package com.mas.sap.prak.client360.candidate;

import com.mas.sap.prak.client360.blocking.BlockingKey;
import com.mas.sap.prak.client360.config.PipelineConfig;
import com.mas.sap.prak.client360.config.SourceConfig;
import com.mas.sap.prak.client360.model.ImportedRecord;

import java.util.*;
import java.util.concurrent.*;
import java.util.stream.Collectors;

public class CandidateGenerator {

    private final PipelineConfig config;

    public CandidateGenerator(
            PipelineConfig config) {

        this.config = config;
    }

    public List<CandidatePair> generate(
            List<BlockingKey> blockingKeys)
            throws Exception {

        Map<String,List<BlockingKey>> blocks =
                blockingKeys.stream()
                        .collect(
                                Collectors.groupingBy(
                                        BlockingKey::getBlockKey));

        ExecutorService executor =
                Executors.newFixedThreadPool(
                        config.getThreading()
                                .getCandidateThreads());

        List<Future<List<CandidatePair>>>
                futures =
                new ArrayList<>();

        for(Map.Entry<String,List<BlockingKey>> entry :
                blocks.entrySet()) {

            futures.add(
                    executor.submit(
                            () -> generateBlock(
                                    entry.getKey(),
                                    entry.getValue())));
        }

        List<CandidatePair> result =
                new ArrayList<>();

        for(Future<List<CandidatePair>> future :
                futures) {

            result.addAll(
                    future.get());
        }

        executor.shutdown();

        return result;
    }

    private List<CandidatePair> generateBlock(
            String blockKey,
            List<BlockingKey> keys) {

        List<CandidatePair> result =
                new ArrayList<>();

        int size = keys.size();

        if(size < config.getBlockingConfig()
                .getMinimumBlockSize()) {

            return result;
        }

        if(size >
                config.getBlockingConfig()
                        .getMaximumBlockSize()) {

            return result;
        }

        for(int i=0;i<size;i++) {

            BlockingKey left =
                    keys.get(i);

            for(int j=i+1;j<size;j++) {

                BlockingKey right =
                        keys.get(j);

                if(left.getEntityId()
                        .equals(
                                right.getEntityId())) {

                    continue;
                }

                result.add(
                        buildPair(
                                left,
                                right));
            }
        }

        return result;
    }

    private CandidatePair buildPair(
            BlockingKey left,
            BlockingKey right) {

        CandidatePair pair =
                new CandidatePair();

        pair.setLeftEntityId(
                left.getEntityId());

        pair.setRightEntityId(
                right.getEntityId());

        pair.setLeftEntityType(
                left.getEntityType());

        pair.setRightEntityType(
                right.getEntityType());

        pair.setBlockKey(
                left.getBlockKey());

        pair.setSourceField(
                left.getSourceField());

        pair.setLeftRecord(
                left.getRecord());

        pair.setRightRecord(
                right.getRecord());

        SourceConfig leftSource =
                left.getSourceConfig();

        SourceConfig rightSource =
                right.getSourceConfig();

        pair.setLeftSearchData(
                extractFields(
                        left.getRecord(),
                        leftSource.getSearchFields()));

        pair.setRightSearchData(
                extractFields(
                        right.getRecord(),
                        rightSource.getSearchFields()));

        pair.setLeftDisplayData(
                extractFields(
                        left.getRecord(),
                        leftSource.getDisplayFields()));

        pair.setRightDisplayData(
                extractFields(
                        right.getRecord(),
                        rightSource.getDisplayFields()));

        pair.setLeftRelationData(
                extractFields(
                        left.getRecord(),
                        leftSource.getRelationFields()));

        pair.setRightRelationData(
                extractFields(
                        right.getRecord(),
                        rightSource.getRelationFields()));

        return pair;
    }

    private Map<String,Object> extractFields(
            ImportedRecord record,
            List<String> fields) {

        Map<String,Object> result =
                new LinkedHashMap<>();

        if(fields == null) {
            return result;
        }

        for(String field : fields) {

            result.put(
                    field,
                    record.get(field));
        }

        return result;
    }
}
