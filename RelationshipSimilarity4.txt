package com.mas.sap.prak.client360.matching;

public class RelationshipSimilarity
        implements SimilarityAlgorithm {

    @Override
    public ComparatorType getType() {

        return ComparatorType.RELATIONSHIP;
    }

    @Override
    public double score(
            String left,
            String right) {

        if(left == null ||
           right == null) {

            return 0.0;
        }

        return left.equalsIgnoreCase(
                right)
                ? 1.0
                : 0.0;
    }
}
