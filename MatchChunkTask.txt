package com.mas.sap.prak.client360.matching;

import com.mas.sap.prak.client360.candidate.CandidatePair;

import java.util.ArrayList;
import java.util.List;
import java.util.concurrent.Callable;

public class MatchChunkTask
        implements Callable<List<MatchResult>> {

    private final List<CandidatePair> pairs;

    private final MatchingEngine engine;

    public MatchChunkTask(
            List<CandidatePair> pairs,
            MatchingEngine engine) {

        this.pairs = pairs;
        this.engine = engine;
    }

    @Override
    public List<MatchResult> call() {

        List<MatchResult> results =
                new ArrayList<>();

        for(CandidatePair pair :
                pairs) {

            results.add(
                    engine.matchPair(
                            pair));
        }

        return results;
    }
}
